% --- Pill badge component ---
% Renders a white-outlined rounded-rectangle badge around inline text.
% Usage:
%   \pill{Version \paperversion}        — default padding (3ex)
%   \pill[4ex]{02. 2026}               — override inner xsep for shorter strings
\newcommand{\pill}[2][3ex]{%
    \begin{tikzpicture}
        \node[
            draw=white,
            line width=0.8pt,
            shape=rounded rectangle,
            inner xsep=#1,
            inner ysep=1.2ex,
            text=white,
            font=\small\sffamily\bfseries,
        ] {#2};
    \end{tikzpicture}%
}
